alias backup='git add . && git commit -m "automated backup" && git push'
alias batcheck='upower -i $(upower -e | grep "BAT") | grep -E "state|to\ full|percentage|capacity|energy-full$|cycle-count"'
alias bye='sync && sleep 1 && sudo shutdown now'
alias checksys='echo "--- BATTERY ---" && batcheck && echo "" && echo "--- DRIVE ---" && drivehealth && echo "" && echo "--- ZRAM ---" && zstat'
alias dashboard='checksys && echo "" && echo "--- REAL-TIME CPU/RAM (Ctrl+C to stop) ---" && top -b -n 1 | head -n 12'
alias drivehealth='sudo smartctl -H /dev/sda && sudo smartctl -A /dev/sda | grep -i "temperature"'
alias fix-swap='sudo swapoff /swapfile && sudo dd if=/dev/zero of=/swapfile bs=1M count=4096 && sudo mkswap /swapfile && sudo swapon /swapfile && sudo modprobe zram && sudo zramctl /dev/zram0 --algorithm zstd --size 4G && sudo mkswap /dev/zram0 && sudo swapon /dev/zram0 --priority 100 && zramctl && free -h'
alias go='sudo sysctl vm.swappiness=100 && echo performance | sudo tee /sys/devices/system/cpu/cpu*/cpufreq/scaling_governor && zstat'
alias grep='grep --color=auto'
alias ls='ls --color=auto'
alias purge='sudo apt-get autoremove -y && sudo apt-get autoclean && sudo find /tmp -type f -atime +2 -delete'
alias znuke='sudo swapoff -a && sudo systemctl restart zramswap && sudo swapon -a && zstat'
alias zstat='zramctl && swapon --show && free -h'
